<Delete Duplicate Emails>
delete from person
where id not in(
select sub.min_id
from(
select email, min(id) as min_id
from person
group by email
) sub)

Delete에 Join 활용하기 // self join 활용 가능
delete t1, t2
from t1
inner join t2 on t1.id = t2.ref
where t1.id = 1;
